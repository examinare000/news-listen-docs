# Tech News Podcast App — Backend Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** GCP 上に RSS 収集・レコメンド計算・Podcast 生成のバッチジョブと、iOS アプリ向け REST API を構築する。

**Architecture:** Cloud Run Jobs × 3（rss-fetcher / recommendation / podcast-generator）が毎日 06:00 JST に Cloud Scheduler から起動。Cloud Run Service（FastAPI）が iOS アプリからのリクエストを処理。Firestore がデータストア、Cloud Storage が音声ファイルの保存先。

**Tech Stack:** Python 3.12, FastAPI, google-genai（Gemini API）, feedparser, trafilatura, google-cloud-firestore, google-cloud-storage, pytest, Docker, gcloud CLI

# backend/shared/models.py
from __future__ import annotations
from dataclasses import dataclass, field
from datetime import datetime, timezone
from enum import Enum
from typing import Optional


class Difficulty(str, Enum):
    TOEIC_600 = "toeic_600"
    TOEIC_900 = "toeic_900"
    IELTS_55 = "ielts_55"
    IELTS_7 = "ielts_7"
    EIKEN_2 = "eiken_2"
    EIKEN_P1 = "eiken_p1"


@dataclass
class RssSource:
    name: str
    url: str

    def to_dict(self) -> dict:
        return {"name": self.name, "url": self.url}

    @classmethod
    def from_dict(cls, d: dict) -> "RssSource":
        return cls(name=d["name"], url=d["url"])


@dataclass
class Article:
    id: str
    title: str
    url: str
    source: str
    content: str
    published_at: datetime
    fetched_at: datetime
    content_fetched_at: Optional[datetime] = None

    def to_dict(self) -> dict:
        return {
            "title": self.title,
            "url": self.url,
            "source": self.source,
            "content": self.content,
            "published_at": self.published_at,
            "fetched_at": self.fetched_at,
            "content_fetched_at": self.content_fetched_at,
        }

    @classmethod
    def from_dict(cls, doc_id: str, d: dict) -> "Article":
        return cls(
            id=doc_id,
            title=d["title"],
            url=d["url"],
            source=d["source"],
            content=d.get("content", ""),
            published_at=d["published_at"],
            fetched_at=d["fetched_at"],
            content_fetched_at=d.get("content_fetched_at"),
        )


@dataclass
class RecommendedArticle:
    article_id: str
    score: float


@dataclass
class Recommendation:
    user_id: str
    date: str  # "2026-05-31"
    articles: list[RecommendedArticle] = field(default_factory=list)
    generated_at: Optional[datetime] = None

    def to_dict(self) -> dict:
        return {
            "user_id": self.user_id,
            "date": self.date,
            "articles": [{"article_id": a.article_id, "score": a.score} for a in self.articles],
            "generated_at": self.generated_at or datetime.now(timezone.utc),
        }

    @classmethod
    def from_dict(cls, d: dict) -> "Recommendation":
        return cls(
            user_id=d["user_id"],
            date=d["date"],
            articles=[
                RecommendedArticle(article_id=a["article_id"], score=a["score"])
                for a in d.get("articles", [])
            ],
            generated_at=d.get("generated_at"),
        )


@dataclass
class UserPrefs:
    user_id: str
    starred_article_ids: list[str] = field(default_factory=list)
    dismissed_article_ids: list[str] = field(default_factory=list)
    rss_sources: list[RssSource] = field(default_factory=lambda: [
        RssSource(name="HackerNews", url="https://hnrss.org/frontpage"),
        RssSource(name="Zenn", url="https://zenn.dev/feed"),
    ])
    default_difficulty: Difficulty = Difficulty.TOEIC_900
    default_playback_speed: float = 1.0
    digest_enabled: bool = True
    digest_article_count: int = 5

    def to_dict(self) -> dict:
        return {
            "user_id": self.user_id,
            "starred_article_ids": self.starred_article_ids,
            "dismissed_article_ids": self.dismissed_article_ids,
            "rss_sources": [s.to_dict() for s in self.rss_sources],
            "default_difficulty": self.default_difficulty.value,
            "default_playback_speed": self.default_playback_speed,
            "digest_enabled": self.digest_enabled,
            "digest_article_count": self.digest_article_count,
        }

    @classmethod
    def from_dict(cls, d: dict) -> "UserPrefs":
        return cls(
            user_id=d["user_id"],
            starred_article_ids=d.get("starred_article_ids", []),
            dismissed_article_ids=d.get("dismissed_article_ids", []),
            rss_sources=[RssSource.from_dict(s) for s in d.get("rss_sources", [])],
            default_difficulty=Difficulty(d.get("default_difficulty", Difficulty.TOEIC_900.value)),
            default_playback_speed=d.get("default_playback_speed", 1.0),
            digest_enabled=d.get("digest_enabled", True),
            digest_article_count=d.get("digest_article_count", 5),
        )


@dataclass
class Podcast:
    id: str
    type: str  # "single" | "digest"
    article_ids: list[str]
    difficulty: Difficulty
    audio_url: str
    japanese_intro_text: str
    duration_seconds: int
    created_at: datetime
    user_id: str

    def to_dict(self) -> dict:
        return {
            "type": self.type,
            "article_ids": self.article_ids,
            "difficulty": self.difficulty.value,
            "audio_url": self.audio_url,
            "japanese_intro_text": self.japanese_intro_text,
            "duration_seconds": self.duration_seconds,
            "created_at": self.created_at,
            "user_id": self.user_id,
        }

    @classmethod
    def from_dict(cls, doc_id: str, d: dict) -> "Podcast":
        return cls(
            id=doc_id,
            type=d["type"],
            article_ids=d["article_ids"],
            difficulty=Difficulty(d["difficulty"]),
            audio_url=d["audio_url"],
            japanese_intro_text=d.get("japanese_intro_text", ""),
            duration_seconds=d.get("duration_seconds", 0),
            created_at=d["created_at"],

# backend/shared/firestore_client.py
from __future__ import annotations
import hashlib
from datetime import datetime, timezone
from google.cloud import firestore
from shared.models import Article, UserPrefs, Recommendation, Podcast, RssSource, Difficulty


class FirestoreClient:
    def __init__(self) -> None:
        self._db = firestore.Client()

    # ---------- Articles ----------

    @staticmethod
    def article_id(url: str) -> str:
        return hashlib.sha256(url.encode()).hexdigest()[:20]

    def article_exists(self, article_id: str) -> bool:
        doc = self._db.collection("articles").document(article_id).get()
        return doc.exists

    def save_article(self, article: Article) -> None:
        self._db.collection("articles").document(article.id).set(
            article.to_dict(), merge=True
        )

    def get_article(self, article_id: str) -> Article | None:
        doc = self._db.collection("articles").document(article_id).get()
        if not doc.exists:
            return None
        return Article.from_dict(doc.id, doc.to_dict())

    def get_recent_articles(self, limit: int = 200) -> list[Article]:
        docs = (
            self._db.collection("articles")
            .order_by("published_at", direction=firestore.Query.DESCENDING)
            .limit(limit)
            .stream()
        )
        return [Article.from_dict(doc.id, doc.to_dict()) for doc in docs]

    # ---------- UserPrefs ----------

    def get_user_prefs(self, user_id: str) -> UserPrefs:
        doc = self._db.collection("userPrefs").document(user_id).get()
        if not doc.exists:
            return UserPrefs(user_id=user_id)
        return UserPrefs.from_dict(doc.to_dict())

    def save_user_prefs(self, prefs: UserPrefs) -> None:
        self._db.collection("userPrefs").document(prefs.user_id).set(prefs.to_dict())

    def add_starred_article(self, user_id: str, article_id: str) -> None:
        self._db.collection("userPrefs").document(user_id).set(
            {"starred_article_ids": firestore.ArrayUnion([article_id])}, merge=True
        )

    def add_dismissed_article(self, user_id: str, article_id: str) -> None:
        self._db.collection("userPrefs").document(user_id).set(
            {"dismissed_article_ids": firestore.ArrayUnion([article_id])}, merge=True
        )

    # ---------- Recommendations ----------

    def save_recommendation(self, rec: Recommendation) -> None:
        doc_id = f"{rec.user_id}_{rec.date}"
        self._db.collection("recommendations").document(doc_id).set(rec.to_dict())

    def get_recommendation(self, user_id: str, date: str) -> Recommendation | None:
        doc_id = f"{user_id}_{date}"
        doc = self._db.collection("recommendations").document(doc_id).get()
        if not doc.exists:
            return None
        return Recommendation.from_dict(doc.to_dict())

    # ---------- Podcasts ----------

    def save_podcast(self, podcast: Podcast) -> None:
        self._db.collection("podcasts").document(podcast.id).set(podcast.to_dict())

    def get_podcasts_for_user(self, user_id: str, limit: int = 50) -> list[Podcast]:
        docs = (
            self._db.collection("podcasts")
            .where("user_id", "==", user_id)
            .order_by("created_at", direction=firestore.Query.DESCENDING)
            .limit(limit)
            .stream()
        )
        return [Podcast.from_dict(doc.id, doc.to_dict()) for doc in docs]

    def podcast_exists_for_article(self, user_id: str, article_id: str, difficulty: str) -> bool:
        docs = list(
            self._db.collection("podcasts")
            .where("user_id", "==", user_id)
            .where("article_ids", "array_contains", article_id)
            .where("difficulty", "==", difficulty)
            .where("type", "==", "single")
            .limit(1)
            .stream()
        )
        return len(docs) > 0
```

# backend/shared/storage_client.py
from __future__ import annotations
import os
from google.cloud import storage


class StorageClient:
    def __init__(self) -> None:
        self._client = storage.Client()
        self._bucket_name = os.environ["GCS_BUCKET_NAME"]

    def upload_audio(self, podcast_id: str, difficulty: str, audio_bytes: bytes) -> str:
        bucket = self._client.bucket(self._bucket_name)
        blob_name = f"podcasts/{podcast_id}/{difficulty}.mp3"
        blob = bucket.blob(blob_name)
        blob.upload_from_string(audio_bytes, content_type="audio/mpeg")
        blob.make_public()
        return blob.public_url

from __future__ import annotations
import os
from google import genai
from google.genai import types


class GeminiClient:
    TEXT_MODEL = "gemini-2.5-flash"
    TTS_MODEL = "gemini-2.5-flash-preview-tts"

    def __init__(self, api_key: str | None = None) -> None:
        key = api_key or os.environ["GEMINI_API_KEY"]
        self._client = genai.Client(api_key=key)

    def generate_text(self, prompt: str, temperature: float = 0.7) -> str:
        response = self._client.models.generate_content(
            model=self.TEXT_MODEL,
            contents=prompt,
            config=types.GenerateContentConfig(temperature=temperature),
        )
        return response.text

    def generate_tts(self, text: str, voice: str = "Kore") -> bytes:
        """テキストを音声バイト列（PCM/WAV）に変換する。
        voice 名: Kore（女性・落ち着いた）, Puck（男性・明瞭）, Charon, Fenrir 等
        """
        response = self._client.models.generate_content(
            model=self.TTS_MODEL,
            contents=text,
            config=types.GenerateContentConfig(
                response_modalities=["AUDIO"],
                speech_config=types.SpeechConfig(
                    voice_config=types.VoiceConfig(
                        prebuilt_voice_config=types.PrebuiltVoiceConfig(voice_name=voice)
                    )
                ),
            ),
        )
        return response.candidates[0].content.parts[0].inline_data.data

# backend/jobs/rss_fetcher/rss_fetcher.py
from __future__ import annotations
import hashlib
from datetime import datetime, timezone
from email.utils import parsedate_to_datetime
import feedparser
from shared.models import Article


class RssFetcher:
    def fetch(self, url: str, source_name: str) -> list[Article]:
        feed = feedparser.parse(url)
        articles = []
        now = datetime.now(timezone.utc)
        for entry in feed.entries:
            link = entry.get("link", "")
            if not link:
                continue
            published_at = self._parse_date(entry) or now
            article = Article(
                id=self.article_id_for(link),
                title=entry.get("title", ""),
                url=link,
                source=source_name,
                content=entry.get("summary", ""),
                published_at=published_at,
                fetched_at=now,
            )
            articles.append(article)
        return articles

    def article_id_for(self, url: str) -> str:
        return hashlib.sha256(url.encode()).hexdigest()[:20]

    def _parse_date(self, entry) -> datetime | None:
        for field in ("published", "updated", "created"):
            raw = entry.get(field)
            if raw:
                try:
                    dt = parsedate_to_datetime(raw)
                    return dt.astimezone(timezone.utc)
                except Exception:
                    pass
        return None
```

# backend/jobs/recommendation/recommender.py
from __future__ import annotations
import json
import logging
from shared.gemini_client import GeminiClient
from shared.models import Article, UserPrefs, RecommendedArticle

logger = logging.getLogger(__name__)

_SCORE_PROMPT = """\
あなたはコンテンツレコメンドエンジンです。ユーザーの過去の行動から関心を分析し、候補記事にスコアを付けてください。

【ユーザーがStarした記事（好み）】
{starred}

【ユーザーがDismissした記事（非好み）】
{dismissed}

【スコアを付ける候補記事】
{candidates}

各候補記事に 0.0〜1.0 のスコアを付けてください。
1.0 = ユーザーが強く興味を持つ、0.0 = 全く興味を持たない。

必ず JSON 配列のみを返してください（他のテキスト不要）:
[{{"id": "記事ID", "score": 0.8}}, ...]
"""


class Recommender:
    def __init__(self, gemini_client: GeminiClient | None = None) -> None:
        self._gemini = gemini_client or GeminiClient()

    def score_articles(
        self,
        prefs: UserPrefs,
        candidates: list[Article],
        starred_articles: list[Article],
    ) -> list[RecommendedArticle]:
        if not candidates:
            return []

        starred_titles = [a.title for a in starred_articles]
        dismissed_titles = []  # dismissed 記事のタイトルは Firestore から別途取得が必要な場合あり

        candidate_data = [{"id": a.id, "title": a.title, "source": a.source} for a in candidates]

        prompt = _SCORE_PROMPT.format(
            starred=json.dumps(starred_titles, ensure_ascii=False) if starred_titles else "なし",
            dismissed=json.dumps(dismissed_titles, ensure_ascii=False) if dismissed_titles else "なし",
            candidates=json.dumps(candidate_data, ensure_ascii=False),
        )

        try:
            raw = self._gemini.generate_text(prompt, temperature=0.2)
            # JSON 部分のみ抽出（```json ブロックに入っている場合に対応）
            if "```" in raw:
                raw = raw.split("```")[1]
                if raw.startswith("json"):
                    raw = raw[4:]
            scored = json.loads(raw.strip())
            return [RecommendedArticle(article_id=s["id"], score=float(s["score"])) for s in scored]
        except Exception as e:
            logger.error(f"Recommendation scoring failed: {e}")
            return [RecommendedArticle(article_id=a.id, score=0.5) for a in candidates]
```

    today = datetime.now(timezone.utc).strftime("%Y-%m-%d")

    db = FirestoreClient()
    gemini = GeminiClient()
    recommender = Recommender(gemini_client=gemini)

    prefs = db.get_user_prefs(user_id)

    scores = recommender.score_articles(prefs, candidates, starred_articles)

    Difficulty.EIKEN_2: "日常〜社会的話題の語彙。標準的な英文。英検2級のリスニング問題レベル。",
    Difficulty.EIKEN_P1: "時事・専門語彙。論説文レベルの複雑な文構造。英検準1級レベル。",
}

_SCRIPT_PROMPT = """\
あなたはポッドキャストの台本ライターです。以下の記事からポッドキャスト台本を作成してください。

【対象記事】
タイトル: {title}
本文:
{content}

【関連記事】
{related}

【難易度指示】
{difficulty_instruction}

【出力フォーマット（必ずこの形式で出力）】
===JAPANESE_INTRO===
（ここに日本語イントロを書く。日付「{date_str}」から始め、記事の概要を1〜5センテンスで紹介する。）

===ENGLISH_BODY===
（ここに英語本編を書く。上記の難易度指示に従うこと。関連記事の内容も自然に組み込むこと。3〜8分相当のテキスト量。）
"""


@dataclass
class PodcastScript:
    japanese_intro: str
    english_body: str


class ScriptGenerator:
    def __init__(self, gemini_client: GeminiClient | None = None) -> None:
        self._gemini = gemini_client or GeminiClient()

    def generate(
        self,
        main_article: Article,
        related_articles: list[Article],
        difficulty: Difficulty,
        date_str: str,
    ) -> PodcastScript:
        related_text = "\n".join(
            f"- {a.title}: {a.content[:300]}" for a in related_articles
        ) or "なし"

        prompt = _SCRIPT_PROMPT.format(
            title=main_article.title,
            content=main_article.content[:3000],
            related=related_text,
            difficulty_instruction=_DIFFICULTY_INSTRUCTIONS[difficulty],
            date_str=date_str,
        )

        raw = self._gemini.generate_text(prompt, temperature=0.8)

        intro = ""
        body = ""
        if "===JAPANESE_INTRO===" in raw and "===ENGLISH_BODY===" in raw:
            parts = raw.split("===ENGLISH_BODY===")
            intro = parts[0].replace("===JAPANESE_INTRO===", "").strip()
            body = parts[1].strip()
        else:
            logger.warning("Script format not found, using raw output as body")
            body = raw

        return PodcastScript(japanese_intro=intro, english_body=body)
```

class TtsGenerator:

# backend/jobs/podcast_generator/main.py
import os
import uuid
import logging
from datetime import datetime, timezone
from shared.firestore_client import FirestoreClient
from shared.gemini_client import GeminiClient
from shared.storage_client import StorageClient
from shared.models import Podcast, Difficulty
from jobs.podcast_generator.script_generator import ScriptGenerator
from jobs.podcast_generator.tts_generator import TtsGenerator

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)


def main() -> None:
    user_id = os.environ.get("USER_ID", "default")
    difficulty_str = os.environ.get("DIFFICULTY", Difficulty.TOEIC_900.value)
    difficulty = Difficulty(difficulty_str)
    today = datetime.now(timezone.utc).strftime("%Y-%m-%d")

    db = FirestoreClient()
    gemini = GeminiClient()
    storage = StorageClient()
    script_gen = ScriptGenerator(gemini_client=gemini)
    tts_gen = TtsGenerator(gemini_client=gemini)

    prefs = db.get_user_prefs(user_id)
    all_articles = db.get_recent_articles(limit=500)
    articles_by_id = {a.id: a for a in all_articles}

    for article_id in prefs.starred_article_ids:
        if db.podcast_exists_for_article(user_id, article_id, difficulty.value):
            logger.info(f"Podcast already exists for article {article_id}, skipping")
            continue

        article = articles_by_id.get(article_id)
        if not article:
            logger.warning(f"Article {article_id} not found in Firestore")
            continue

        # 関連記事: 同じソースの最新5件（簡易実装）
        related = [
            a for a in all_articles
            if a.source == article.source and a.id != article.id
        ][:5]

        logger.info(f"Generating podcast for: {article.title}")
        try:
            script = script_gen.generate(article, related, difficulty, today)
            audio_bytes = tts_gen.generate_audio(script)

            podcast_id = str(uuid.uuid4())
            audio_url = storage.upload_audio(podcast_id, difficulty.value, audio_bytes)

            # 音声長さを概算（PCM 24kHz モノラル想定: bytes / 24000 / 2）
            duration = len(audio_bytes) // 48000

            podcast = Podcast(
                id=podcast_id,
                type="single",
                article_ids=[article_id],
                difficulty=difficulty,
                audio_url=audio_url,
                japanese_intro_text=script.japanese_intro,
                duration_seconds=duration,
                created_at=datetime.now(timezone.utc),
                user_id=user_id,
            )
            db.save_podcast(podcast)
            logger.info(f"Saved podcast {podcast_id} for article {article_id}")
        except Exception as e:
            logger.error(f"Failed to generate podcast for {article_id}: {e}")
            continue


if __name__ == "__main__":
    main()
```

    db = FirestoreClient()

                title=article.title,
                url=article.url,
                source=article.source,
                score=scores_by_id.get(article.id, 0.5),
                published_at=article.published_at.isoformat(),
            ))
